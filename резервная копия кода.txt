enum ItemType { accommodation, service, custom }

class BookingItem {
  String id;
  String title;
  String description;
  int price;
  String unit;
  String image;
  List<String> images;
  ItemType type;
  int capacity;

  BookingItem({
    required this.id,
    required this.title,
    this.description = '',
    required this.price,
    this.unit = '',
    this.image = '',
    this.images = const [],
    required this.type,
    this.capacity = 2,
  });
}

    required this.startDate,
    required this.endDate,
    required this.totalPrice,
    required this.prepayment,

    this.userId = '',

  Future<void> _initUserId() async {
    final prefs = await SharedPreferences.getInstance();
    currentUserId = prefs.getString('device_user_id') ?? '';
    if (currentUserId.isEmpty) {
      // Если человек зашел впервые - придумываем ему ID
      currentUserId = 'user_${DateTime.now().millisecondsSinceEpoch}';
      await prefs.setString('device_user_id', currentUserId);
    }
    notifyListeners();

  // --- НАЧАЛО: Метод для проверки занятости дня в календаре ---
  Map<String, dynamic> getDayStatus(DateTime day) {
    final checkDay = DateTime(day.year, day.month, day.day);

    // Список ID домиков, которые заняты в этот день
    List<String> bookedItemsIds = [];

    // Перебираем все бронирования
    for (var t in _transactions) {
      DateTime s = DateTime(t.startDate.year, t.startDate.month, t.startDate.day);
      DateTime e = DateTime(t.endDate.year, t.endDate.month, t.endDate.day);

      // Если день попадает в диапазон брони (от заезда до выезда не включительно)
      if ((checkDay.isAtSameMomentAs(s) || checkDay.isAfter(s)) && checkDay.isBefore(e)) {
        bookedItemsIds.add(t.itemId);
      }
    }

    // Считаем общую вместимость (сколько всего домиков)
    int totalCapacity = accommodations.fold(0, (sum, item) => sum + item.capacity);
    int totalBooked = bookedItemsIds.length;

    return {
      'bookedCount': totalBooked,
      'totalCapacity': totalCapacity,
      'isFull': totalCapacity > 0 && totalBooked >= totalCapacity, // Заняты ли все доступные места
      'bookedIds': bookedItemsIds,
    };
  }

  // Получаем доходы за конкретный месяц и год
  double monthlyIncomes(int month, int year) {
    return incomes
        .where((i) => i.date.month == month && i.date.year == year)
        .fold(0.0, (sum, item) => sum + item.amount);
  }

  // Получаем расходы за конкретный месяц и год
  double monthlyExpenses(int month, int year) {
    return expenses
        .where((e) => e.date.month == month && e.date.year == year)
        .fold(0.0, (sum, item) => sum + item.amount);
  }

  // Чистая прибыль за месяц
  double monthlyNetProfit(int month, int year) =>
      monthlyIncomes(month, year) - monthlyExpenses(month, year);

  // Этот метод заставляет приложение "слушать" базу данных постоянно
  void listenToTransactions() {
    // 1. Слушаем доходы (с защитой от кривых дат)
    _incomesSub = FirebaseFirestore.instance
        .collection('${prefix}incomes')
        .snapshots()
        .listen((snapshot) {
          incomes.clear();
          for (var doc in snapshot.docs) {
            final data = doc.data();
            incomes.add(
              Income(
                id: doc.id,
                title: data['title'] ?? '',
                amount: (data['amount'] as num?)?.toInt() ?? 0,
                date: DateTime.tryParse(data['date'] ?? '') ?? DateTime.now(),
                category: data['category'] ?? 'Общее',
              ),
            );
          }
          notifyListeners();
        });

    // 2. Слушаем расходы (с защитой от кривых дат)
    _expensesSub = FirebaseFirestore.instance
        .collection('${prefix}expenses')
        .snapshots()
        .listen((snapshot) {
          expenses.clear();
          for (var doc in snapshot.docs) {
            final data = doc.data();
            expenses.add(
              Expense(
                id: doc.id,
                title: data['title'] ?? '',
                amount: (data['amount'] as num?)?.toInt() ?? 0,
                date: DateTime.tryParse(data['date'] ?? '') ?? DateTime.now(),
                category: data['category'] ?? 'Общее',
              ),
            );
          }
          notifyListeners();
        });

    // 3. БРОНИРОВАНИЯ (БРОНЕЖИЛЕТ ОТ ОШИБОК)
    _bookingsSub = FirebaseFirestore.instance
        .collection('${prefix}bookings')
        .snapshots()
        .listen((snapshot) {
          final List<BookingTransaction> validTransactions = [];

          for (var doc in snapshot.docs) {
            final data = doc.data();
            try {
              // Проверяем, что в базе есть все нужные поля, иначе игнорируем бронь
              if (data['startDate'] != null &&
                  data['endDate'] != null &&
                  data['itemId'] != null) {
                validTransactions.add(
                  BookingTransaction(
                    itemId: data['itemId'].toString(),
                    startDate: DateTime.parse(data['startDate']),
                    endDate: DateTime.parse(data['endDate']),
                    userId: data['userId'] ?? '',
                  ),
                );
              }
            } catch (e) {
              // Если старая бронь кривая, мы её просто игнорируем,
              // чтобы она НЕ ЛОМАЛА весь остальной календарь!
              debugPrint('Пропущена кривая бронь: ${doc.id}');
            }
          }

          _transactions = validTransactions;
          notifyListeners(); // Обновляем календарь при новых бронях
        });
  }

  Future<void> confirmBooking(String bookingId, int prepaymentAmount) async {
    final firestore = FirebaseFirestore.instance;

    try {
      // 1. ИЩЕМ БРОНЬ НАПРЯМУЮ В ОБЛАКЕ (чтобы работало для браузера)
      DocumentSnapshot bookingSnap = await firestore
          .collection('${prefix}bookings')
          .doc(bookingId)
          .get();

      if (!bookingSnap.exists) {
        print("❌ Бронь не найдена в базе");
        return;
      }

      // Вытаскиваем данные для красивого сообщения в Телеграм
      final data = bookingSnap.data() as Map<String, dynamic>;
      String itemTitle = data['itemTitle'] ?? 'Неизвестный объект';
      String customerName = data['customerName'] ?? 'Без имени';

      // 2. ОБНОВЛЯЕМ СТАТУС И ПРЕДОПЛАТУ В ОБЛАКЕ
      await firestore.collection('${prefix}bookings').doc(bookingId).update({
        'isConfirmed': true,
        'prepayment': prepaymentAmount,
      });

      // 3. Обновляем локально в памяти телефона, если бронь там есть
      final index = bookings.indexWhere((b) => b.id == bookingId);
      if (index != -1) {
        bookings[index].isConfirmed = true;
        bookings[index].prepayment = prepaymentAmount;
        _saveToDisk();
      }

      notifyListeners(); // Перерисовываем экран (галочка появится мгновенно)

      // 4. Отправляем красивое уведомление в Телеграм
      sendTelegramMessage(
        '✅ *Бронь подтверждена!*\n'
        '🏠 Объект: $itemTitle\n'
        '👤 Клиент: $customerName\n'
        '💰 Внесена предоплата: $prepaymentAmount ₽',
      );

      print("✅ Бронь $bookingId успешно подтверждена!");
    } catch (e) {
      print("❌ Ошибка при подтверждении: $e");
    }
  }

  Future<void> removeBooking(String id) async {
    final firestore = FirebaseFirestore.instance;

    try {
      // 1. ИЩЕМ БРОНЬ НАПРЯМУЮ В ОБЛАКЕ (чтобы удалять даже те, что созданы с браузера/другого телефона)
      DocumentSnapshot bookingSnap = await firestore
          .collection('${prefix}bookings')
          .doc(id)
          .get();

      if (!bookingSnap.exists) {
        // Если в облаке брони уже нет, просто чистим локальный кэш
        bookings.removeWhere((b) => b.id == id);
        incomes.removeWhere((i) => i.id == id);
        notifyListeners();
        return;
      }

      // Вытаскиваем данные из облачного документа
      final data = bookingSnap.data() as Map<String, dynamic>;
      String itemId = data['itemId'];
      DateTime startDate = DateTime.parse(data['startDate']);
      DateTime endDate = DateTime.parse(data['endDate']);
      String customerName = data['customerName'] ?? 'Без имени';
      String itemTitle = data['itemTitle'] ?? 'Неизвестный объект';

      // 2. ЗАПУСКАЕМ ТРАНЗАКЦИЮ ДЛЯ СНЯТИЯ ЗАМКА
      await firestore.runTransaction((transaction) async {
        DocumentReference lockRef = firestore
            .collection('${prefix}calendar_locks')
            .doc(itemId);
        DocumentSnapshot lockSnapshot = await transaction.get(lockRef);

        if (lockSnapshot.exists && lockSnapshot.data() != null) {
          final lockData = lockSnapshot.data() as Map<String, dynamic>;
          Map<String, int> occupancyMap = {};
          lockData.forEach(
            (key, value) => occupancyMap[key] = (value as num).toInt(),
          );

          // Высчитываем даты, которые нужно освободить
          List<String> daysToRemove = [];
          DateTime current = DateTime(
            startDate.year,
            startDate.month,
            startDate.day,
          );
          DateTime end = DateTime(endDate.year, endDate.month, endDate.day);

          while (current.isBefore(end)) {
            daysToRemove.add(
              "${current.year}-${current.month.toString().padLeft(2, '0')}-${current.day.toString().padLeft(2, '0')}",
            );
            current = current.add(const Duration(days: 1));
          }
          if (daysToRemove.isEmpty) {
            daysToRemove.add(
              "${current.year}-${current.month.toString().padLeft(2, '0')}-${current.day.toString().padLeft(2, '0')}",
            );
          }

          // Вычитаем даты из замка
          for (String day in daysToRemove) {
            if (occupancyMap.containsKey(day)) {
              occupancyMap[day] = occupancyMap[day]! - 1;
              if (occupancyMap[day]! <= 0) {
                occupancyMap.remove(
                  day,
                ); // Стираем дату полностью, если мест больше не занято
              }
            }
          }

          // Обновляем замок в базе
          transaction.set(lockRef, occupancyMap);
        }

        // 3. Удаляем саму бронь и доход из облака
        transaction.delete(firestore.collection('${prefix}bookings').doc(id));
        transaction.delete(firestore.collection('${prefix}incomes').doc(id));
      });

      // 4. Удаляем из памяти телефона и обновляем экран
      bookings.removeWhere((b) => b.id == id);
      incomes.removeWhere((i) => i.id == id);
      _saveToDisk();
      notifyListeners();

      // 5. Отправляем уведомление об отмене в Телеграм
      sendTelegramMessage(
        '❌ *Бронь отменена!*\n'
        '🏠 Объект: $itemTitle\n'
        '👤 Клиент: $customerName',
      );

      print("✅ Бронь удалена, замок снят!");
    } catch (e) {
      print("❌ Ошибка при удалении брони: $e");
    }
  }

  void updateItemPrice(String id, int newPrice) {
    // Ищем в домиках
    for (var item in accommodations) {
      if (item.id == id) {
        item.price = newPrice;
        sendTelegramMessage(
          '⚠️ *Изменение цены на ${item.title}!*\n💰 Новая цена: $newPrice ₽',
        );
        break;
      }
    }
    // Ищем в услугах
    for (var item in services) {
      if (item.id == id) {
        item.price = newPrice;
        break;
      }
    }
    FirebaseFirestore.instance
        .collection('${prefix}accommodations')
        .doc(id)
        .update({'price': newPrice});

    _saveToDisk(); // Сохраняем, чтобы цена не слетела после перезагрузки
    notifyListeners(); // Обновляем экран
  }

  // --- ФУНКЦИЯ СОХРАНЕНИЯ (Записывает всё на диск) ---
  Future<void> _saveToDisk() async {
    final prefs = await SharedPreferences.getInstance();
    // 2. Сохраняем бронирования

  // --- ФУНКЦИЯ ЗАГРУЗКИ (Читает с диска при старте) ---
  Future<void> _loadFromDisk() async {
    final prefs = await SharedPreferences.getInstance();
    final String? savedPrices = prefs.getString('saved_prices');
    if (savedPrices != null) {
      final List decodedPrices = json.decode(savedPrices);
      for (var priceData in decodedPrices) {
        // Ищем домик в списке и обновляем ему цену
        for (var item in accommodations) {
          if (item.id == priceData['id']) {
            item.price = priceData['price'];
          }
        }
      }
    }
    // Загружаем бронирования
    final String? savedBookings = prefs.getString('saved_bookings');
    if (savedBookings != null) {
      final List decodedB = json.decode(savedBookings);
      bookings = decodedB.map((data) {
        // Проверяем: если ID 'manual', создаем заглушку прямо на лету
        BookingItem item;
        if (data['itemId'] == 'manual') {
          item = BookingItem(
            id: 'manual',
            title: data['customerName'] ?? 'Ручной доход',
            description: '',
            price: 0,
            unit: '',
            type: ItemType.custom,
          );
        } else {
          // Иначе ищем в списке домиков или услуг
          item = accommodations.firstWhere(
            (a) => a.id == data['itemId'],
            orElse: () => services.firstWhere(
              (s) => s.id == data['itemId'],
              orElse: () => BookingItem(
                id: 'unknown',
                title: 'Удалено',
                description: '',
                price: 0,
                unit: '',
                type: ItemType.custom,
              ),
            ),
          );
        }

        return Booking(
          id: data['id'],
          customerName: data['customerName'],
          customerPhone: data['customerPhone'] ?? '',
          item: item,
          startDate: DateTime.parse(data['startDate']),
          endDate: DateTime.parse(data['endDate']),
          totalPrice: data['totalPrice'],
          prepayment: data['prepayment'] ?? 0,
          isConfirmed: data['isConfirmed'] ?? false,
          userId: data['userId'] ?? '',
        );
      }).toList();
    }
    final String? savedServices = prefs.getString('saved_custom_services');
    if (savedServices != null) {
      final List decodedS = json.decode(savedServices);
      services = decodedS
          .map(
            (data) => BookingItem(
              id: data['id'],
              title: data['title'],
              description: data['description'],
              price: data['price'],
              unit: data['unit'],
              type: ItemType.service,
              image: data['image'] ?? '',
            ),
          )
          .toList();
    }
    notifyListeners();
  }

  Future<bool> addBooking(Booking booking) async {
    final firestore = FirebaseFirestore.instance;

    try {
      // 1. Берем лимит мест из самого объекта (1 или 2)
      int limit = booking.item.capacity;

      // ЗАПУСКАЕМ ТРАНЗАКЦИЮ
      await firestore.runTransaction((transaction) async {
        // 2. Ссылка на ДОКУМЕНТ-ЗАМОК (Один документ на каждый домик)
        DocumentReference lockRef = firestore
            .collection('${prefix}calendar_locks')
            .doc(booking.item.id);

        // ВАЖНО: Читаем именно этот документ внутри транзакции
        DocumentSnapshot lockSnapshot = await transaction.get(lockRef);

        // Карта занятости: {"2024-03-15": 1} (дата и сколько мест занято)
        Map<String, int> occupancyMap = {};
        if (lockSnapshot.exists && lockSnapshot.data() != null) {
          final data = lockSnapshot.data() as Map<String, dynamic>;
          data.forEach((key, value) => occupancyMap[key] = value as int);
        }

        // 3. Считаем даты бронирования
        List<String> daysToBook = [];
        DateTime current = DateTime(
          booking.startDate.year,
          booking.startDate.month,
          booking.startDate.day,
        );
        DateTime end = DateTime(
          booking.endDate.year,
          booking.endDate.month,
          booking.endDate.day,
        );

        while (current.isBefore(end)) {
          daysToBook.add(
            "${current.year}-${current.month.toString().padLeft(2, '0')}-${current.day.toString().padLeft(2, '0')}",
          );
          current = current.add(const Duration(days: 1));
        }
        if (daysToBook.isEmpty)
          daysToBook.add(
            "${current.year}-${current.month.toString().padLeft(2, '0')}-${current.day.toString().padLeft(2, '0')}",
          );

        // 4. ПРОВЕРКА МЕСТ: Если в любой из дней мест >= лимиту, отменяем всё
        for (String day in daysToBook) {
          int currentlyBooked = occupancyMap[day] ?? 0;
          if (currentlyBooked >= limit) {
            throw Exception(
              "overlap",
            ); // Бросаем ошибку, транзакция прервется автоматически
          }
        }

        // 5. Если места есть — обновляем карту
        for (String day in daysToBook) {
          occupancyMap[day] = (occupancyMap[day] ?? 0) + 1;
        }

        // 6. ПИШЕМ ВСЁ В БАЗУ ОДНИМ ПАКЕТОМ
        transaction.set(lockRef, occupancyMap); // Обновляем замок

        DocumentReference bookingRef = firestore
            .collection('${prefix}bookings')
            .doc(booking.id);
        transaction.set(bookingRef, {
          'id': booking.id,
          'customerName': booking.customerName,
          'customerPhone': booking.customerPhone,
          'itemId': booking.item.id,
          'itemTitle': booking.item.title,
          'startDate': booking.startDate.toIso8601String(),
          'endDate': booking.endDate.toIso8601String(),
          'totalPrice': booking.totalPrice,
          'prepayment': booking.prepayment,
          'isConfirmed': booking.isConfirmed,
          'userId': booking.userId,
          'timestamp': FieldValue.serverTimestamp(),
        });

        DocumentReference incRef = firestore
            .collection('${prefix}incomes')
            .doc(booking.id);
        transaction.set(incRef, {
          'id': booking.id,
          'title': 'Бронирование: ${booking.customerName}',
          'amount': booking.totalPrice,
          'date': booking.startDate.toIso8601String(),
          'category': 'Проживание',
        });
      });

      // --- ЕСЛИ ТРАНЗАКЦИЯ УСПЕШНА ---
      bookings.add(booking);
      bookings.sort((a, b) => b.startDate.compareTo(a.startDate));

      final newIncome = Income(
        id: booking.id,
        title: 'Бронирование: ${booking.customerName}',
        amount: booking.totalPrice,
        date: booking.startDate,
        category: 'Проживание',
      );
      incomes.add(newIncome);

      _saveToDisk();
      notifyListeners();

      sendTelegramMessage(
        '🌲 *Новая бронь!*\n\n'
        '🏠 *Объект:* ${booking.item.title}\n'
        '👤 *Клиент:* ${booking.customerName}\n'
        '📞 *Тел:* ${booking.customerPhone}\n'
        '💰 *Сумма:* ${booking.totalPrice} ₽',
      );

      return true;
    } catch (e) {
      print("Ошибка при бронировании: $e");
      return false;
    }
  }

  Future<bool> uploadPhotoToAccommodation(
    String itemId,
    XFile pickedFile,
  ) async {
    try {
      // Твой личный ключ ImgBB
      const String apiKey = "0dd5acfb6502f372d500c8f8e22e2f6e";

      // 1. Читаем файл в байты (работает и на телефоне, и в Chrome)
      final Uint8List fileBytes = await pickedFile.readAsBytes();

      // 2. Готовим запрос к API ImgBB
      var request = http.MultipartRequest(
        'POST',
        Uri.parse('https://api.imgbb.com/1/upload?key=$apiKey'),
      );

      // Добавляем файл
      request.files.add(
        http.MultipartFile.fromBytes(
          'image',
          fileBytes,
          filename: 'photo.jpg', // Мы всегда шлем как jpg, как ты и просил
        ),
      );

      // 3. Отправляем
      var streamedResponse = await request.send();
      var response = await http.Response.fromStream(streamedResponse);

      if (response.statusCode == 200) {
        var jsonResponse = json.decode(response.body);
        // Берем прямую ссылку на картинку
        String downloadUrl = jsonResponse['data']['url'];

        // 4. Записываем ссылку в базу данных Firebase
        await FirebaseFirestore.instance
            .collection('${prefix}accommodations')
            .doc(itemId)
            .update({
              'images': FieldValue.arrayUnion([downloadUrl]),
            });

        // 5. Обновляем экран в приложении
        final index = accommodations.indexWhere((a) => a.id == itemId);
        if (index != -1) {
          accommodations[index].images = List.from(accommodations[index].images)
            ..add(downloadUrl);
          notifyListeners();
        }
        return true;
      } else {
        print("Ошибка ImgBB: ${response.body}");
        return false;
      }
    } catch (e) {
      print("Ошибка при выполнении запроса: $e");
      return false;
    }
  }

  // Вспомогательная функция для прямого скачивания с Яндекс.Диска
  Future<void> _startDirectDownload(BuildContext context, String yandexDiskUrl) async {
    // Показываем крутилку загрузки, чтобы турист понял, что процесс пошел
    showDialog(
      context: context,
      barrierDismissible: false,
      builder: (context) => const Center(child: CircularProgressIndicator(color: Colors.green)),
    );

    try {
      // Магия: просим у Яндекса ПРЯМАЮ ссылку на скачивание файла
      final apiUrl = 'https://cloud-api.yandex.net/v1/disk/public/resources/download?public_key=$yandexDiskUrl';
      final response = await http.get(Uri.parse(apiUrl));

      if (!context.mounted) return;
      Navigator.pop(context); // Убираем крутилку загрузки

      if (response.statusCode == 200) {
        final data = json.decode(response.body);
        final directUrl = data['href']; // Вот она, чистая ссылка на .apk!

        // Телефон мгновенно начнет скачивание файла без открытия сайтов
        await launchUrl(Uri.parse(directUrl), mode: LaunchMode.externalApplication);
      } else {
        // План Б: если Яндекс тупит, открываем по старинке
        await launchUrl(Uri.parse(yandexDiskUrl), mode: LaunchMode.externalApplication);
      }
    } catch (e) {
      if (!context.mounted) return;
      Navigator.pop(context);
      await launchUrl(Uri.parse(yandexDiskUrl), mode: LaunchMode.externalApplication);
    }
  }

  Future<void> _checkForUpdates(BuildContext context) async {
    // 1. ИГНОРИРУЕМ ВЕБ-ВЕРСИЮ: Если приложение запущено в браузере (PWA, iPhone, Telegram),
    // просто прерываем проверку. Им ничего скачивать не нужно.
    if (kIsWeb) return;

    try {
      // 2. Узнаем текущую версию установленного приложения
      PackageInfo packageInfo = await PackageInfo.fromPlatform();
      String currentVersion = packageInfo.version
          .trim(); // Убираем случайные пробелы

      // 3. Спрашиваем Firebase.
      // ВАЖНО: GetOptions(source: Source.server) заставляет код игнорировать
      // старый кэш телефона и брать только свежую версию из облака!
      var doc = await FirebaseFirestore.instance
          .collection('app_config')
          .doc('settings')
          .get(const GetOptions(source: Source.server));

      if (doc.exists) {
        String latestVersion = (doc.data()?['latest_version'] ?? currentVersion)
            .toString()
            .trim();
        String updateUrl = doc.data()?['update_url'] ?? '';

        // 4. Если версии не совпадают — показываем окно
        if (currentVersion != latestVersion && updateUrl.isNotEmpty) {
          if (!mounted) return;
          showDialog(
            context: context,
            barrierDismissible: false, // Запрещаем закрывать окно кликом мимо
            builder: (context) => AlertDialog(
              title: const Text('Доступно обновление! 🚀'),
              content: Text(
                'Вышла новая версия Сказки Алтая ($latestVersion).\nПожалуйста, обновите приложение для стабильной работы.',
              ),
              actions: [
                // Кнопка "СКАЧАТЬ"
                ElevatedButton(
                  onPressed: () {
                    // Вызываем нашу новую функцию для прямого скачивания
                    _startDirectDownload(context, updateUrl);
                  },
                  style: ElevatedButton.styleFrom(
                    backgroundColor: Colors.green,
                  ),
                  child: const Text(
                    'СКАЧАТЬ',
                    style: TextStyle(color: Colors.white),
                  ),
                ),
                // ДОБАВЛЕНО: Кнопка "Позже" на всякий случай, чтобы приложение
                // не блокировалось намертво, если вдруг что-то пойдет не так
                TextButton(
                  onPressed: () => Navigator.pop(context),
                  child: const Text(
                    'Позже',
                    style: TextStyle(color: Colors.grey),
                  ),
                ),
              ],
            ),
          );
        }
      }
    } catch (e) {
      debugPrint('Ошибка проверки обновлений: $e');
    }
  }

class BookingDialog extends StatefulWidget {
  final BookingItem? item; // Добавили знак вопроса
  final bool isAdmin;

  const BookingDialog({
    super.key,
    this.item, // Убрали required
    this.isAdmin = false,
  });

  @override
  State<BookingDialog> createState() => _BookingDialogState();
}

class _BookingDialogState extends State<BookingDialog> {
  final _nameController = TextEditingController();
  final _phoneController = TextEditingController();

  // Переменные для управления выбором
  BookingItem? _selectedItem;
  DateTimeRange? _selectedDateRange;
  int _totalPrice = 0;

  Future<void> _launchURL(String urlString) async {
    final Uri url = Uri.parse(urlString);
    if (!await launchUrl(url, mode: LaunchMode.externalApplication)) {
      throw Exception('Не удалось открыть ссылку $urlString');
    }
  }

  @override
  void initState() {
    super.initState();
    _selectedItem = widget.item;
    // Если айтем есть — ставим его цену, если нет (нажали плюс) — ставим 0
    _totalPrice = _selectedItem?.price ?? 0;
  }

  // Функция пересчета цены
  void _calculatePrice() {
    // Выбираем: либо то, что выбрали в списке, либо то, что пришло изначально
    final currentItem = _selectedItem ?? widget.item;

    if (_selectedDateRange != null && currentItem != null) {
      int days = _selectedDateRange!.duration.inDays;
      if (days == 0) days = 1;

      setState(() {
        _totalPrice = currentItem.price * days;
      });
    }
  }

  bool _isDayBlocked(DateTime day) {
    final provider = Provider.of<DataBaseProvider>(context, listen: false);
    if (_selectedItem == null) return false;

    // Считаем, сколько броней уже есть на этот КОНКРЕТНЫЙ ID объекта
    final existingBookingsCount = provider.transactions.where((t) {
      if (t.itemId != _selectedItem!.id) return false;

      final start = DateTime(t.startDate.year, t.startDate.month, t.startDate.day);
      final end = DateTime(t.endDate.year, t.endDate.month, t.endDate.day);
      final check = DateTime(day.year, day.month, day.day);

      // Проверяем занятость ночи
      return (check.isAtSameMomentAs(start) || (check.isAfter(start) && check.isBefore(end)));
    }).length;

    // Сравниваем с лимитом объекта (для Верхнего/Нижнего это будет 2)
    return existingBookingsCount >= _selectedItem!.capacity;
  }

  void _pickDateRange() async {
    final provider = Provider.of<DataBaseProvider>(context, listen: false);
    final now = DateTime.now();

    // 1. ИСПРАВЛЕНИЕ: Берем объект из выпадающего списка (если он выбран),
    // либо изначальный объект. Это убирает ошибку "Сначала выберите объект".
    final currentItem = _selectedItem ?? widget.item;

    if (currentItem == null || currentItem.id == 'manual') {
      ScaffoldMessenger.of(context).showSnackBar(
        const SnackBar(content: Text('Сначала выберите домик или услугу!')),
      );
      return;
    }

    final newDateRange = await showDateRangePicker(
      context: context,
      firstDate: now,
      lastDate: DateTime(now.year + 1),
      initialDateRange: _selectedDateRange,

      // 2. ИСПРАВЛЕНИЕ КАЛЕНДАРЯ: Предикат должен принимать ровно ОДИН параметр (day).
      selectableDayPredicate: (DateTime day, DateTime? startRange, DateTime? endRange) {
        if (provider.seasonStart != null && provider.seasonEnd != null) {
          DateTime checkDay = DateTime(day.year, day.month, day.day);
          DateTime sStart = DateTime(provider.seasonStart!.year, provider.seasonStart!.month, provider.seasonStart!.day);
          DateTime sEnd = DateTime(provider.seasonEnd!.year, provider.seasonEnd!.month, provider.seasonEnd!.day);

          if (checkDay.isBefore(sStart) || checkDay.isAfter(sEnd)) return false;
        }

        if (currentItem.type != ItemType.accommodation) return true;
        DateTime checkDay = DateTime(day.year, day.month, day.day);

        for (var t in provider.transactions) {
          DateTime startT = DateTime(t.startDate.year, t.startDate.month, t.startDate.day);
          DateTime endT = DateTime(t.endDate.year, t.endDate.month, t.endDate.day);

          // ⚠️ ГЛАВНОЕ ИЗМЕНЕНИЕ ЗДЕСЬ:
          // Убрали checkDay.isAtSameMomentAs(startT).
          // День заезда (startT) остается кликабельным, чтобы служить днем выезда для других!
          if (checkDay.isAfter(startT) && checkDay.isBefore(endT)) {

            // --- ЛОГИКА ВЗАИМОСВЯЗЕЙ ---
            if (t.itemId == 'combo_base') return false;
            if (currentItem.id == 'combo_base') return false;

            if (t.itemId == 'combo_teremok') {
              if (currentItem.id == 'combo_teremok') return false;
              if (currentItem.title.toLowerCase().contains('теремок')) return false;
            }

            if (currentItem.id == 'combo_teremok') {
              bool isAnyRoomInTeremokBooked = provider.accommodations.any((a) => a.id == t.itemId && a.title.toLowerCase().contains('теремок'));
              if (isAnyRoomInTeremokBooked) return false;
            }

            if (t.itemId == currentItem.id) return false;
          }
        }
        return true;
      },

      // 3. ИСПРАВЛЕНИЕ ЦВЕТА: Красим занятые (заблокированные) даты в красный
      builder: (context, child) {
        return Theme(
          data: Theme.of(context).copyWith(
            datePickerTheme: DatePickerThemeData(
              // 1. Делаем сам текст белым на красном фоне
              dayForegroundColor: WidgetStateProperty.resolveWith((states) {
                if (states.contains(WidgetState.disabled)) return Colors.white;
                return Colors.black87;
              }),
              // 2. ЗАЛИВАЕМ ФОН ЗАНЯТЫХ ДАТ КРАСНЫМ
              dayBackgroundColor: WidgetStateProperty.resolveWith((states) {
                if (states.contains(WidgetState.disabled))
                  return Colors.red.shade400;
                return null;
              }),
            ),
          ),
          child: child!,
        );
      },
    );

    if (newDateRange != null) {
      bool hasOverlap = false;
      DateTime current = newDateRange.start;
      // Создаем отдельную переменную для конца проверки
      DateTime endRange = newDateRange.end;

      // --- ИСПРАВЛЕНИЕ: Если выбрали одни сутки (с 17 по 17) ---
      // Добавляем 1 день, чтобы цикл обязательно проверил эту ночь!
      if (current.isAtSameMomentAs(endRange)) {
        endRange = endRange.add(const Duration(days: 1));
      }

      // Проверяем каждую ночь в выбранном диапазоне
      while (current.isBefore(endRange)) {
        DateTime checkDay = DateTime(current.year, current.month, current.day);

        for (var t in provider.transactions) {
          DateTime startT = DateTime(t.startDate.year, t.startDate.month, t.startDate.day);
          DateTime endT = DateTime(t.endDate.year, t.endDate.month, t.endDate.day);

          // Проверяем занята ли эта ночь (включая день заезда, но не включая день выезда)
          if ((checkDay.isAtSameMomentAs(startT) || checkDay.isAfter(startT)) && checkDay.isBefore(endT)) {

            if (t.itemId == 'combo_base' || currentItem.id == 'combo_base') {
              hasOverlap = true; break;
            }
            if (t.itemId == 'combo_teremok' && (currentItem.id == 'combo_teremok' || currentItem.title.toLowerCase().contains('теремок'))) {
              hasOverlap = true; break;
            }
            if (currentItem.id == 'combo_teremok' && t.itemId != 'combo_teremok' && provider.accommodations.any((a) => a.id == t.itemId && a.title.toLowerCase().contains('теремок'))) {
              hasOverlap = true; break;
            }
            if (t.itemId == currentItem.id) {
              hasOverlap = true; break;
            }
          }
        }
        if (hasOverlap) break;
        current = current.add(const Duration(days: 1));
      }

      // Если нашли пересечение — выдаем ошибку и отменяем выбор
      if (hasOverlap) {
        ScaffoldMessenger.of(context).showSnackBar(
          const SnackBar(
            content: Text('Выбранные даты пересекаются с другой бронью! Выберите свободные дни.'),
            backgroundColor: Colors.red,
            duration: Duration(seconds: 4),
          ),
        );
        return; // 🔴 Бронь не проходит дальше
      }

      // Если всё свободно, сохраняем выбор и считаем цену
      setState(() {
        _selectedDateRange = newDateRange;
      });
      _calculatePrice();
    }
  }

          ),
          ElevatedButton(
            onPressed: () {

              Provider.of<DataBaseProvider>(
                context,
                listen: false,
              ).confirmBooking(docId, prepAmount);
              Navigator.pop(ctx);
            },
            style: ElevatedButton.styleFrom(backgroundColor: Colors.green),
            child: const Text(
              'Подтвердить',
              style: TextStyle(color: Colors.white),
            ),
          ),
        ],
      ),
    );
  }

  @override
  Widget build(BuildContext context) {
    // 1. СНАЧАЛА получаем доступ к провайдеру!
    final provider = Provider.of<DataBaseProvider>(context);

    // 2. Теперь используем provider.prefix вместо просто prefix
    final Stream<QuerySnapshot> _bookingsStream = FirebaseFirestore.instance
        .collection('${provider.prefix}bookings') // <--- ИСПРАВЛЕНИЕ ЗДЕСЬ
        .orderBy('timestamp', descending: true)
        .snapshots();

    return StreamBuilder<QuerySnapshot>(
      stream: _bookingsStream,
      builder: (BuildContext context, AsyncSnapshot<QuerySnapshot> snapshot) {
        // Если произошла ошибка
        if (snapshot.hasError) {
          return const Center(child: Text('Что-то пошло не так'));
        }

        // Пока данные загружаются
        if (snapshot.connectionState == ConnectionState.waiting) {
          return const Center(child: CircularProgressIndicator());
        }

        // Если броней в облаке вообще нет
        if (snapshot.data!.docs.isEmpty) {
          return const Center(child: Text('Бронирований в облаке пока нет'));
        }

        // 2. Если данные есть, строим список
        // 2. Если данные есть, строим список
        return ListView(
          children: snapshot.data!.docs.map((DocumentSnapshot document) {
            Map<String, dynamic> data =
                document.data()! as Map<String, dynamic>;
            String docId = document.id;

            // --- НОВОЕ: Форматируем даты для вывода на экран ---
            String dateText = "Даты не указаны";
            if (data['startDate'] != null && data['endDate'] != null) {
              DateTime sDate = DateTime.parse(data['startDate']);
              DateTime eDate = DateTime.parse(data['endDate']);
              dateText =

  // Список операций (исправляем 4 ошибки 'expense')
  Widget _buildTransactionList(DataBaseProvider provider) {
    // 1. Берем УЖЕ ГОТОВЫЕ списки из провайдера (они там всегда актуальны благодаря listenToTransactions)
    // Фильтруем по выбранному месяцу
    final monthIncomes = provider.incomes
        .where(
          (i) =>
              i.date.year == _selectedMonth.year &&
              i.date.month == _selectedMonth.month,
        )
        .toList();

    final monthExpenses = provider.expenses
        .where(
          (e) =>
              e.date.year == _selectedMonth.year &&
              e.date.month == _selectedMonth.month,
        )
        .toList();

    // Соединяем в один список для истории
    final List<Map<String, dynamic>> items = [
      ...monthIncomes.map((d) => {'doc': d, 'type': 'income'}),
      ...monthExpenses.map((d) => {'doc': d, 'type': 'expense'}),
    ];

    // Сортируем: новые сверху
    items.sort((a, b) => (b['doc'].date).compareTo(a['doc'].date));

    if (items.isEmpty) {
      return const Center(child: Text("Нет операций за этот месяц"));
    }

    return ListView.builder(
      // КРИТИЧЕСКИЕ НАСТРОЙКИ ДЛЯ СКРОЛЛА:
      shrinkWrap: true,
      physics: const NeverScrollableScrollPhysics(),
      itemCount: items.length,
      itemBuilder: (context, index) {
        final item = items[index];
        final data = item['doc']; // Это уже объект Income или Expense
        final bool isIncomeItem = item['type'] == 'income';

        return Dismissible(
          key: Key(data.id),
          // Используем уникальный ID из базы
          direction: DismissDirection.endToStart,
          background: Container(
            color: Colors.red,
            alignment: Alignment.centerRight,
            padding: const EdgeInsets.only(right: 20),
            child: const Icon(Icons.delete, color: Colors.white),
          ),
          onDismissed: (direction) {
            // Просто вызываем удаление, провайдер сам обновит экран
            if (isIncomeItem) {
              provider.deleteIncome(data.id);
            } else {
              provider.deleteExpense(data.id);
            }
          },
          child: ListTile(
            title: Text(data.title),
            subtitle: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Text(DateFormat('dd.MM.yyyy').format(data.date)),
                // Ищем предоплату прямо из списка броней, не трогая саму математику дохода
                if (isIncomeItem &&
                    provider.bookings.any(
                      (b) => b.id == data.id && b.prepayment > 0,
                    ))
                  Text(
                    'В т.ч. предоплата: ${provider.bookings.firstWhere((b) => b.id == data.id).prepayment} ₽',
                    style: const TextStyle(
                      color: Colors.blueGrey,
                      fontSize: 12,
                    ),
                  ),
              ],
            ),
            trailing: Text(
              '${isIncomeItem ? "+" : "-"}${data.amount} ₽',
              // Сумма остается полной!
              style: TextStyle(
                color: isIncomeItem ? Colors.green : Colors.red,
                fontWeight: FontWeight.bold,
              ),
            ),
          ),
        );
      },
    );
  }
}

class FinancialPieChart extends StatelessWidget {
  final double income;
  final double expense;

  const FinancialPieChart({
    super.key,
    required this.income,
    required this.expense,
  });

  @override
  Widget build(BuildContext context) {
    // 1. Используем .abs(), чтобы точно не было отрицательных чисел (из-за них часто летит ошибка)
    final double safeIncome = income.abs();
    final double safeExpense = expense.abs();

    // 2. Если всё по нулям, просто выводим текст, не запуская диаграмму
    if (safeIncome == 0 && safeExpense == 0) {
      return const Padding(
        padding: EdgeInsets.symmetric(vertical: 30),
        child: Center(
          child: Text(
            "Нет данных за этот период",
            style: TextStyle(color: Colors.grey),
          ),
        ),
      );
    }

    return Container(
      margin: const EdgeInsets.symmetric(vertical: 10, horizontal: 16),
      padding: const EdgeInsets.all(20),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(20),
        boxShadow: [
          BoxShadow(color: Colors.black.withOpacity(0.05), blurRadius: 10),
        ],
      ),
      child: Column(
        children: [
          const Text(
            "Статистика",
            style: TextStyle(fontWeight: FontWeight.bold, fontSize: 18),
          ),
          const SizedBox(height: 20),
          SizedBox(
            height: 150,
            child: PieChart(
              PieChartData(
                sectionsSpace: 2,
                centerSpaceRadius: 40,
                // 3. Рисуем секцию только если значение больше 0
                sections: [
                  if (safeIncome > 0)
                    PieChartSectionData(
                      value: safeIncome,
                      title: '',
                      color: Colors.greenAccent[700],
                      radius: 20,
                    ),
                  if (safeExpense > 0)
                    PieChartSectionData(
                      value: safeExpense,
                      title: '',
                      color: Colors.redAccent,
                      radius: 20,
                    ),
                ],
              ),
            ),
          ),
          const SizedBox(height: 20),
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceAround,
            children: [
              _indicator(
                Colors.greenAccent[700]!,
                "Доходы",
                "${safeIncome.toInt()} ₽",
              ),
              _indicator(
                Colors.redAccent,
                "Расходы",
                "${safeExpense.toInt()} ₽",
              ),
            ],
          ),
        ],
      ),
    );
  }

  Widget _indicator(Color color, String label, String amount) {
    return Column(
      children: [
        Row(
          children: [
            Container(
              width: 8,
              height: 8,
              decoration: BoxDecoration(color: color, shape: BoxShape.circle),
            ),
            const SizedBox(width: 4),
            Text(
              label,
              style: const TextStyle(fontSize: 12, color: Colors.grey),
            ),
          ],
        ),
        Text(amount, style: const TextStyle(fontWeight: FontWeight.bold)),
      ],
    );
  }

class MonthAdminView extends StatelessWidget {
  final DateTime month;

  const MonthAdminView({super.key, required this.month});

  @override
  Widget build(BuildContext context) {
    final provider = Provider.of<DataBaseProvider>(context);
    // Определяем количество дней в месяце
    int daysInMonth = DateTime(month.year, month.month + 1, 0).day;

    return Scaffold(
      appBar: AppBar(title: Text(DateFormat('MMMM yyyy', 'ru').format(month))),
      body: GridView.builder(
        padding: const EdgeInsets.all(10),
        gridDelegate: const SliverGridDelegateWithFixedCrossAxisCount(
          crossAxisCount: 7, // 7 дней недели
          mainAxisSpacing: 8,
          crossAxisSpacing: 8,
        ),
        itemCount: daysInMonth,
        itemBuilder: (context, index) {
          final day = DateTime(month.year, month.month, index + 1);
          final status = provider.getDayStatus(day);

          Color bgColor = Colors.green.shade100;
          if (status['bookedCount'] > 0) bgColor = Colors.orange.shade100;
          if (status['isFull']) bgColor = Colors.red.shade100;

          return GestureDetector(
            onTap: () =>
                _showDayDetails(context, day, status['bookedIds'], provider),
            child: Container(
              decoration: BoxDecoration(
                color: bgColor,
                borderRadius: BorderRadius.circular(8),
                border: Border.all(
                  color: status['bookedCount'] > 0
                      ? Colors.orange
                      : Colors.green,
                ),
              ),
              child: Column(
                mainAxisAlignment: MainAxisAlignment.center,
                children: [
                  Text(
                    '${index + 1}',
                    style: const TextStyle(fontWeight: FontWeight.bold),
                  ),
                  Text(
                    '${status['bookedCount']}/${status['totalCapacity']}',
                    style: const TextStyle(fontSize: 10),
                  ),
                ],
              ),
            ),
          );
        },
      ),
    );
  }

  void _showDayDetails(
    BuildContext context,
    DateTime day,
    List<String> bookedIds,
    DataBaseProvider provider,
  ) {
    showDialog(
      context: context,
      builder: (context) => AlertDialog(
        title: Text('Загрузка на ${DateFormat('dd.MM').format(day)}'),
        content: Column(
          mainAxisSize: MainAxisSize.min,
          children: provider.accommodations.map((item) {
            // Считаем сколько раз этот конкретный домик забронирован сегодня
            int count = bookedIds.where((id) => id == item.id).length;
            return ListTile(
              title: Text(item.title),
              trailing: Text('$count / ${item.capacity}'),
              subtitle: Text(
                count >= item.capacity ? 'Мест нет' : 'Есть свободные номера',
              ),
              leading: Icon(
                Icons.home,
                color: count >= item.capacity ? Colors.red : Colors.green,
              ),
            );
          }).toList(),
        ),
        actions: [
          TextButton(
            onPressed: () => Navigator.pop(context),
            child: const Text('Закрыть'),
          ),
        ],
      ),
    );
  }
}

// --- НОВАЯ Вкладка Отчеты (Аналитика и Налоги) ---
class AdminReportsTab extends StatefulWidget {
  const AdminReportsTab({super.key});

  @override
  State<AdminReportsTab> createState() => _AdminReportsTabState();
}

class _AdminReportsTabState extends State<AdminReportsTab> {
  int _selectedYear = DateTime.now().year;
  int _selectedTaxMonth = DateTime.now().month;

  @override
  Widget build(BuildContext context) {
    final provider = Provider.of<DataBaseProvider>(context);

    // --- ЛОГИКА ДЛЯ НАЛОГОВОЙ (ОНА РАБОТАЛА, ТАК КАК ДОХОДЫ ИДУТ ИЗ FIREBASE) ---
    final taxIncome = provider.incomes
        .where(
          (i) =>
              i.date.year == _selectedYear && i.date.month == _selectedTaxMonth,
        )
        .fold(0.0, (sum, item) => sum + item.amount);

    return SingleChildScrollView(
      padding: const EdgeInsets.all(16.0),
      child: Column(
        crossAxisAlignment: CrossAxisAlignment.start,
        children: [
          // ВЫБОР ГОДА (Общий для всего экрана)
          Row(
            mainAxisAlignment: MainAxisAlignment.spaceBetween,
            children: [
              const Text(
                'Аналитика за:',
                style: TextStyle(fontSize: 18, fontWeight: FontWeight.bold),
              ),
              DropdownButton<int>(
                value: _selectedYear,
                items: [2024, 2025, 2026, 2027]
                    .map(
                      (y) => DropdownMenuItem(value: y, child: Text('$y год')),
                    )
                    .toList(),
                onChanged: (val) => setState(() => _selectedYear = val!),
              ),
            ],
          ),
          const Divider(),

          // БЛОК 1: ДЛЯ НАЛОГОВОЙ
          Container(
            padding: const EdgeInsets.all(16),
            decoration: BoxDecoration(
              color: Colors.blue.shade50,
              borderRadius: BorderRadius.circular(15),
              border: Border.all(color: Colors.blue.shade200),
            ),
            child: Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                Row(
                  mainAxisAlignment: MainAxisAlignment.spaceBetween,
                  children: [
                    const Text(
                      'Сводка для "Мой Налог"',
                      style: TextStyle(
                        fontWeight: FontWeight.bold,
                        fontSize: 16,
                      ),
                    ),
                    DropdownButton<int>(
                      value: _selectedTaxMonth,
                      items: List.generate(12, (index) => index + 1).map((m) {
                        return DropdownMenuItem(
                          value: m,
                          child: Text(
                            DateFormat('MMMM', 'ru').format(DateTime(2000, m)),
                          ),
                        );
                      }).toList(),
                      onChanged: (val) =>
                          setState(() => _selectedTaxMonth = val!),
                    ),
                  ],
                ),
                const SizedBox(height: 10),
                const Text(
                  'Сумма поступлений за месяц (без вычета расходов):',
                  style: TextStyle(color: Colors.blueGrey, fontSize: 12),
                ),
                Text(
                  '${taxIncome.toInt()} ₽',
                  style: const TextStyle(
                    fontSize: 28,
                    fontWeight: FontWeight.bold,
                    color: Colors.blue,
                  ),
                ),
                const SizedBox(height: 10),
                Text(
                  '* Эту сумму необходимо задекларировать в приложении ФНС',
                  style: TextStyle(
                    color: Colors.grey.shade600,
                    fontSize: 11,
                    fontStyle: FontStyle.italic,
                  ),
                ),
              ],
            ),
          ),
          const SizedBox(height: 20),

          // БЛОК 2: РЕЙТИНГ ПРИБЫЛЬНОСТИ ЖИЛЬЯ ЗА ГОД
          const Text(
            'Выручка по объектам за год',
            style: TextStyle(fontWeight: FontWeight.bold, fontSize: 16),
          ),
          const SizedBox(height: 10),

          // --- ИСПРАВЛЕНИЕ: ЧИТАЕМ ДАННЫЕ НАПРЯМУЮ ИЗ FIREBASE ---
          StreamBuilder<QuerySnapshot>(
            stream: FirebaseFirestore.instance
                .collection('${provider.prefix}bookings')
                .snapshots(),
            builder: (context, snapshot) {
              if (snapshot.connectionState == ConnectionState.waiting) {
                return const Center(child: CircularProgressIndicator());
              }

              // Создаем мапу: {ID домика: Общая заработанная сумма}
              Map<String, int> itemProfits = {};
              for (var a in provider.accommodations) {
                itemProfits[a.id] = 0; // Инициализируем нулями
              }

              if (snapshot.hasData) {
                for (var doc in snapshot.data!.docs) {
                  final data = doc.data() as Map<String, dynamic>;
                  try {
                    bool isConfirmed = data['isConfirmed'] ?? false;
                    if (isConfirmed && data['startDate'] != null) {
                      DateTime sDate = DateTime.parse(data['startDate']);

                      // Считаем только за выбранный год
                      if (sDate.year == _selectedYear) {
                        String itemId = data['itemId'];
                        int price = data['totalPrice'] ?? 0;

                        if (itemProfits.containsKey(itemId)) {
                          itemProfits[itemId] = itemProfits[itemId]! + price;
                        }
                      }
                    }
                  } catch (e) {
                    // Игнорируем кривые брони
                  }
                }
              }

              // Сортируем домики по прибыльности (от большего к меньшему)
              var sortedItems = itemProfits.entries.toList()
                ..sort((a, b) => b.value.compareTo(a.value));

              return Column(
                children: sortedItems.map((entry) {
                  // Находим название домика по ID
                  String itemName = provider.accommodations
                      .firstWhere(
                        (a) => a.id == entry.key,
                        orElse: () => BookingItem(
                          id: '',
                          title: 'Удаленный объект',
                          price: 0,
                          type: ItemType.accommodation,
                        ),
                      )
                      .title;
                  int profit = entry.value;

                  return Card(
                    margin: const EdgeInsets.only(bottom: 8),
                    child: ListTile(
                      leading: CircleAvatar(
                        backgroundColor: profit > 0
                            ? Colors.green.shade100
                            : Colors.grey.shade200,
                        child: Icon(
                          Icons.house,
                          color: profit > 0 ? Colors.green : Colors.grey,
                        ),
                      ),
                      title: Text(
                        itemName,
                        style: const TextStyle(fontWeight: FontWeight.bold),
                      ),
                      trailing: Text(
                        '$profit ₽',
                        style: TextStyle(
                          color: profit > 0 ? Colors.green : Colors.grey,
                          fontWeight: FontWeight.bold,
                          fontSize: 16,
                        ),
                      ),
                    ),
                  );
                }).toList(),
              );
            },
          ),
        ],
      ),
    );
  }
}
